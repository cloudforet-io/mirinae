## Manual Control

<Canvas>
    <Story name="Manual Control">
        {{
            components: {
                PLazyImg,
                PButton
            },
            template: `
    <div class="flex items-center">
        <p-button class="mr-4" @click="getData">Load Image</p-button>
        <p-lazy-img :src="src"
            :loading="loading"
        />
    </div>
`,
            setup() {
                const state = reactive({
                    src: '',
                    loading: false
                })
                const getData = () => {
                    state.src = '';
                    state.loading = true
                    setTimeout(() => {
                        state.src = faker.image.image(100, 100, true)
                        state.loading = false;
                    }, 2000)
                }
                return {
                    ...toRefs(state),
                    getData
                }
            }
        }}
    </Story>
</Canvas>
